function doFirst(){ let canvas = document.getElementById('canvas'); let context = canvas.getContext('2d'); // 劃格線 context.beginPath(); for(var i=0; i<25; i++){ var position = i * 50; context.moveTo(position,0); context.lineTo(position,canvas.height); context.fillText(position,position,10); context.moveTo(0,position); context.lineTo(canvas.width,position); context.fillText(position,0,position); } context.strokeStyle = 'rgba(0,0,0,0.3)'; context.stroke(); // 劃出四分之一
